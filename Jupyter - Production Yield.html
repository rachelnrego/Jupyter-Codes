# Build data frame
df_pareto = pd.DataFrame({"Process": [99,86,25,64,35,42,22,55,38,10,7,19]})

# Reset the indexes
df_pareto.index = ["Poor Training","Irrigation","Leakages of Spraying Equipment","Absence of Crop Rotation","Expired Pesticide",
            "Banned Pesticide","Improper disposal of Remaining Pesticide","Labour","Fertilization","Seeding Issues","Sowing",
            "Packaging"]

# Sort values in descending order
df_pareto = df_pareto.sort_values(by='Process', ascending=False)

# Add cumulative percentage column
df_pareto["cum_percentage"] = round(df_pareto["Process"].cumsum()/df_pareto["Process"].sum()*100,2)
