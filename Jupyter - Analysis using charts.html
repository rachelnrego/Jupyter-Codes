import seaborn as sns
sns.set_theme(style="whitegrid")
df2= df.groupby('region').sum()['homeless'].reset_index()

# Draw a nested barplot by Market and Product Type
g = sns.catplot(
    data=df2, kind="bar", x="region", y="homeless",ci="sd", palette="dark", alpha=.6, height=6,order=df2.sort_values('homeless').region
)
plt.xlabel('Region', fontsize=12, color="black")
plt.ylabel('Homeless', fontsize=12, color="black")
plt.title("Region wise Homeless People", fontsize=18, color="black")
g.despine(left=True)
plt.tick_params(axis='x', which='major',rotation=90, labelsize=10)
plt.show()
